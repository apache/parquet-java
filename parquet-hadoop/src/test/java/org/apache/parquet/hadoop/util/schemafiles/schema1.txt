message sortsol_record {
  optional int32 schema_id;
  optional group msg {
    optional double Latitude;
    optional double Longitude;
    optional group FreeShapeDeliveryZoneInfo {
      optional group HexagonIDs (LIST) {
        repeated binary array (UTF8);
      }
      optional group MultipleGeofence (LIST) {
        repeated group array {
          optional group HexagonIDs (LIST) {
            repeated binary array (UTF8);
          }
        }
      }
    }
    optional group FulfillmentZoneInfo {
      optional group DeliveryZoneInfoMap (MAP) {
        repeated group map (MAP_KEY_VALUE) {
          optional group value {
            optional binary UUID (UTF8);
            optional double MaxDeliveryRadiusMiles;
            optional group FreeShapeDeliveryZoneInfo {
              optional group HexagonIDs (LIST) {
                repeated binary array (UTF8);
              }
              optional group MultipleGeofence (LIST) {
                repeated group array {
                  optional group HexagonIDs (LIST) {
                    repeated binary array (UTF8);
                  }
                }
              }
            }
          }
        }
      }
      optional group DeliveryZoneInfoList (LIST) {
        repeated group array {
          optional group FreeShapeDeliveryZoneInfo {
            optional group HexagonIDs (LIST) {
              repeated binary array (UTF8);
            }
            optional group MultipleGeofence (LIST) {
              repeated group array {
                optional group HexagonIDs (LIST) {
                  repeated binary array (UTF8);
                }
              }
            }
          }
        }
      }
    }
  }
}
